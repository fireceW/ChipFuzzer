.section .text
.global _start

_start:
    # 通过构造两种不同符号位的数值来覆盖 io_msb(最高位)相关逻辑
    lui     t0, 0x80000    # t0 = 0x80000000 (RV32) / 0xFFFFFFFF80000000 (RV64),MSB=1
    slt     t1, t0, zero   # 使用符号比较,t1=1(t0为负数),触发对MSB的使用
    lui     t0, 0          # t0 = 0x00000000,MSB=0
    slt     t2, t0, zero   # 使用符号比较,t2=0(t0为非负),再次触发对MSB的使用

    //下面为汇编程序退出逻辑,请不要修改
    li      gp,1
    li      a7,93
    li      a0,0
    ecall
    unimp
    .insn   2, 0x0000
